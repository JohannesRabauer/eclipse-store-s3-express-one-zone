#ENV variables to set
```
# Config for express S3 Bucket
# Example: myBucket--eun1-az1--x-s3
S3_EXPRESS_BUCKET=bucketname--region--zone--x-s3   
S3_EXPRESS_REGION=region

# Config for "normal" S3 Bucket
S3_DEFAULT_BUCKET=bucketname
S3_DEFAULT_REGION=region

# Config both buckets
S3_ACCESS_KEY=accessKey;
S3_SECRET_ACCESS_KEY=secretAccessKey
```
